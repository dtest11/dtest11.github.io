func Bfs(root Node){
    if root == nil {
        return 
    }
    var queue Queue
    queue.Push(root)
    for !queue.IsEmpty(){
       size :=queue.Size()
       for i :=0; i<size; i++{
           cur :=queue.Pop()
           if cur.Left !=nil {
               queue.Push(cur.Left)
           }
           if cur.Right !=nil {
               queue.Push(cur.Right)
           }
       }
    }
}
